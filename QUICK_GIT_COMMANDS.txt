# Quick Git Commands for arset
# Copy and paste these commands one by one in PowerShell or Command Prompt

# Step 1: Navigate to project
cd "C:\Users\arset\Fourfold Fate"

# Step 2: Initialize git (if not already done)
git init

# Step 3: Add all files
git add .

# Step 4: Commit
git commit -m "Initial setup: Core systems, UI, and scene setup"

# Step 5: Add remote 
# Repository already exists at: https://github.com/AndreFerreira23/Fourfold-Fate
# Run this command:
git remote add origin https://github.com/AndreFerreira23/Fourfold-Fate.git

# If you get "remote already exists" error, remove it first:
# git remote remove origin
# Then run the add command again

# Step 6: Push to GitHub
git branch -M main
git push -u origin main

# IMPORTANT: Before running Step 5, create the repository on GitHub:
# 1. Go to https://github.com/new
# 2. Repository name: Fourfold-Fate
# 3. Choose Private or Public
# 4. DON'T check "Initialize with README"
# 5. Click "Create repository"
# 6. Then run Step 5 and 6 above
